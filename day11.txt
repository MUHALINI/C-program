
//stack(push operation)//

#include<stdio.h>
#define size 5
int arr[size];
int top = -1;
//push
void push(int value){
    if(top == size -1){
        printf("No space is left \n");
    }
    else{
        top++;
        arr[top] = value;
        printf("%d value pushed to the array \n", value);
    }
}
int main(){
    push (10);
    push (20);
    push (40);
    push (90);
    push(30);
    push(80);
    return 0;
}

void pop(){
    if(top == -1){
        printf("no value is added at stack");
    }else{
        printf("The value is %d: \n", arr[top]);
        top--;
    }
    
}
