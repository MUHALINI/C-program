#include <stdio.h>
//creating structure
struct structure{
    int rollno;
char section;
float cgpa;
};
int main() {
//create structure variable for the structure
struct structure stru1;
//struct structure student2;
//assiging values for structure variable
stru1.rollno= 30;
stru1.section='a';
stru1.cgpa = 8.9;
//printing values
printf("The Roll no is ; %d \n",stru1.rollno);
printf("The section is : %c \n", stru1.section);
printf("The cgpa is ; %f \n", stru1.cgpa);

    return 0;
}
